Router-BH

en
conf t
host Router-BH
no ip domain-lookup
inter ser 0/1/0
desc LINK-Router-VR
ip address 192.168.0.18 255.255.255.252
clock rate 2000000
no shut
exit
inter g 0/0/1
no shut
exi
inter g 0/0/1.99
encapsulation dot1q 99
no ip addre
exit
inter g 0/0/1.199
encapsulation dot1q 199
ip add 10.0.0.13 255.255.255.0
des LAN-MANAGEMENT
no sh
inter g 0/0/1.500
encapsulation dot1q 500
desc GATEWAY-LAN-BA
ip add 172.21.0.1 255.255.255.0
exi
ip dhcp pool LAN-BH
netw 172.21.0.0 255.255.255.0
dns-server 172.17.2.2
default-router 172.21.0.1
exit
ip dhcp excluded-addr 172.21.0.1 172.21.0.50
router ospf 1
router-id 7.7.7.7
netw 192.168.0.16 0.0.0.3 area 0
netw 10.0.0.0 0.0.0.255 area 0
net 172.21.0.0 0.0.0.255 area 1
exi
do wr
